Create a very simple playbook name is ec2-change-state.yml
add:
 ```
 ---
- hosts: localhost
  gather_facts: yes
  vars_files:
    - keys.yml

  tasks:
  - pip:
      name: boto

  - name: Change instances state by tag
    ec2:
      aws_access_key: "{{ AWS_ACCESS_KEY_ID }}"
      aws_secret_key: "{{ AWS_SECRET_ACCESS_KEY }}"
      region: "{{ AWS_REGION }}"
      state: running
      instance_tags:
           Name: Linux1

```

### Creat ec2-change-state-shell.yml


```
---
- hosts: localhost
  gather_facts: yes
  tasks:
  - name: Change instances state by tag
    local_action: ec2
    args:
       state: running
       instance_tags:
          Name: Linux1


```
